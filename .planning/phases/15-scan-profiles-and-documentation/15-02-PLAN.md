    Add profile JS functions:
    ```javascript
    function showNewProfileForm() {
        document.getElementById('new-profile-card').style.display = 'block';
        document.getElementById('new-profile-name').focus();
    }

    function hideNewProfileForm() {
        document.getElementById('new-profile-card').style.display = 'none';
        document.getElementById('new-profile-name').value = '';
        document.getElementById('new-profile-desc').value = '';
        var checkboxes = document.querySelectorAll('#new-profile-scanners input[type="checkbox"]');
        checkboxes.forEach(function(cb) { cb.checked = false; });
        var timeouts = document.querySelectorAll('#new-profile-scanners input[type="number"]');
        timeouts.forEach(function(t) { t.value = ''; });
    }

    function _collectScanners(prefix) {
        var scanners = {};
        var checkboxes = document.querySelectorAll('[id^="' + prefix + 'chk-"]');
        checkboxes.forEach(function(cb) {
            if (cb.checked) {
                var name = cb.value;
                var timeoutInput = document.getElementById(prefix.replace('chk-', 'timeout-') + name);
                var entry = {};
                if (timeoutInput && timeoutInput.value) {
                    entry.timeout = parseInt(timeoutInput.value, 10);
                }
                scanners[name] = entry;
            }
        });
        return scanners;
    }

    function saveNewProfile() {
        var name = document.getElementById('new-profile-name').value.trim();
        var desc = document.getElementById('new-profile-desc').value.trim();
        if (!name) {
            showAlert('profiles-alert', 'Profile name is required.', 'error');
            return;
        }
        var scanners = _collectScanners('new-');
        if (Object.keys(scanners).length === 0) {
            showAlert('profiles-alert', 'Profile must have at least one scanner enabled. Select at least one scanner and try again.', 'error');
            return;
        }

        fetch('/api/config/profiles', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            credentials: 'same-origin',
            body: JSON.stringify({ name: name, description: desc, scanners: scanners })
        }).then(function(res) {
            if (res.ok) {
                showAlert('profiles-alert', 'Profile saved successfully.', 'success');
                setTimeout(function() { switchTab('profiles'); location.reload(); }, 1000);
            } else {
                return res.json().then(function(data) {
                    showAlert('profiles-alert', data.detail || 'Failed to save profile.', 'error');
                });
            }
        }).catch(function(err) {
            showAlert('profiles-alert', 'Failed to save profile: ' + err.message + '. Check file permissions on config.yml.', 'error');
        });
    }

    function toggleProfileCard(event, profileName) {
        if (event.target.closest('input') || event.target.closest('.btn')) return;
        var allSettings = document.querySelectorAll('[id^="profile-settings-"]');
        allSettings.forEach(function(el) {
            if (el.id !== 'profile-settings-' + profileName) {
                el.classList.remove('card-settings--open');
            }
        });
        var settings = document.getElementById('profile-settings-' + profileName);
        if (settings) { settings.classList.toggle('card-settings--open'); }
    }

    function collapseProfileCard(profileName) {
        var settings = document.getElementById('profile-settings-' + profileName);
        if (settings) { settings.classList.remove('card-settings--open'); }
    }

    function updateProfile(profileName) {
        var desc = document.getElementById('edit-desc-' + profileName).value.trim();
        var scanners = {};
        var checkboxes = document.querySelectorAll('[id^="edit-chk-' + profileName + '-"]');
        checkboxes.forEach(function(cb) {
            if (cb.checked) {
                var name = cb.value;
                var timeoutInput = document.getElementById('edit-timeout-' + profileName + '-' + name);
                var entry = {};
                if (timeoutInput && timeoutInput.value) {
                    entry.timeout = parseInt(timeoutInput.value, 10);
                }
                scanners[name] = entry;
            }
        });

        if (Object.keys(scanners).length === 0) {
            showAlert('profile-alert-' + profileName, 'Profile must have at least one scanner enabled. Select at least one scanner and try again.', 'error');
            return;
        }

        fetch('/api/config/profiles/' + profileName, {
            method: 'PUT',
            headers: { 'Content-Type': 'application/json' },
            credentials: 'same-origin',
            body: JSON.stringify({ description: desc, scanners: scanners })
        }).then(function(res) {
            if (res.ok) {
                showAlert('profile-alert-' + profileName, 'Profile saved successfully.', 'success');
                setTimeout(function() { collapseProfileCard(profileName); }, 1500);
            } else {
                return res.json().then(function(data) {
                    showAlert('profile-alert-' + profileName, data.detail || 'Failed to save profile.', 'error');
                });
            }
        }).catch(function(err) {
            showAlert('profile-alert-' + profileName, 'Failed to save profile: ' + err.message + '. Check file permissions on config.yml.', 'error');
        });
    }

    function deleteProfile(profileName) {
        if (!confirm("Are you sure you want to delete the profile '" + profileName + "'? This cannot be undone.")) return;

        fetch('/api/config/profiles/' + profileName, {
            method: 'DELETE',
            credentials: 'same-origin',
        }).then(function(res) {
            if (res.ok) {
                showAlert('profiles-alert', 'Profile deleted successfully.', 'success');
                var card = document.querySelector('[data-profile="' + profileName + '"]');
                if (card) card.remove();
            } else {
                return res.json().then(function(data) {
                    showAlert('profiles-alert', data.detail || 'Failed to delete profile.', 'error');
                });
            }
        }).catch(function(err) {
            showAlert('profiles-alert', 'Failed to delete profile: ' + err.message + '. Check file permissions on config.yml.', 'error');
        });
    }
    ```
