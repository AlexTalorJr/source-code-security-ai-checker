Create `Makefile` (MUST use literal tab characters for recipe lines, NOT spaces) with these exact contents:

**Variables:**
```makefile
VERSION := $(shell python3 -c "import tomllib; print(tomllib.load(open('pyproject.toml','rb'))['project']['version'])" 2>/dev/null || grep -m1 'version' pyproject.toml | cut -d'"' -f2)
PACKAGE_NAME := aipix-security-scanner-$(VERSION)
REGISTRY ?= localhost
IMAGE_NAME ?= aipix-security-scanner
BACKUP_DIR := backups
TIMESTAMP := $(shell date +%Y%m%d_%H%M%S)

.DEFAULT_GOAL := help
```

**Targets (all .PHONY):**

`.PHONY: help install run stop test migrate backup restore package clean docker-multiarch docker-push`

- `help: ## Show available targets` -- uses grep/awk self-documenting pattern from RESEARCH.md
- `install: ## Build Docker images` -- `docker compose build`
- `run: ## Start scanner (detached)` -- `docker compose up -d`
- `stop: ## Stop scanner` -- `docker compose down`
- `test: ## Run test suite` -- `python -m pytest tests/ -v`
- `migrate: ## Run database migrations` -- `docker compose exec scanner alembic upgrade head`
- `backup: ## Backup DB, reports, and config to timestamped archive` -- Implementation:
  1. `@mkdir -p $(BACKUP_DIR)`
  2. `@echo "Creating backup..."`
  3. Use sqlite3 .backup for safe WAL-mode backup: `docker compose exec -T scanner sqlite3 /data/scanner.db ".backup '/tmp/scanner_backup.db'"`
  4. `docker compose cp scanner:/tmp/scanner_backup.db $(BACKUP_DIR)/scanner.db`
  5. Copy config.yml from host: `cp config.yml $(BACKUP_DIR)/config.yml 2>/dev/null || true`
  6. Copy reports/ directory from container: `docker compose cp scanner:/data/reports $(BACKUP_DIR)/reports 2>/dev/null || mkdir -p $(BACKUP_DIR)/reports`
  7. Create timestamped tar.gz: `tar czf $(BACKUP_DIR)/backup-$(TIMESTAMP).tar.gz -C $(BACKUP_DIR) scanner.db config.yml reports`
  8. Cleanup temp files: `@rm -f $(BACKUP_DIR)/scanner.db $(BACKUP_DIR)/config.yml && rm -rf $(BACKUP_DIR)/reports`
  9. `@echo "Backup saved: $(BACKUP_DIR)/backup-$(TIMESTAMP).tar.gz"`
- `restore: ## Restore from backup (usage: make restore BACKUP=path/to/file.tar.gz)` -- Implementation:
  1. Guard: `@test -n "$(BACKUP)" || (echo "Usage: make restore BACKUP=path/to/file.tar.gz"; exit 1)`
  2. `@test -f "$(BACKUP)" || (echo "File not found: $(BACKUP)"; exit 1)`
  3. `@echo "Restoring from $(BACKUP)..."`
  4. Extract to temp dir: `@mkdir -p /tmp/scanner-restore && tar xzf $(BACKUP) -C /tmp/scanner-restore`
  5. Copy DB back: `docker compose cp /tmp/scanner-restore/scanner.db scanner:/data/scanner.db`
  6. Copy config if present: `@cp /tmp/scanner-restore/config.yml ./config.yml 2>/dev/null || true`
  7. Copy reports if present: `docker compose cp /tmp/scanner-restore/reports scanner:/data/reports 2>/dev/null || true`
  8. Run migrations: `docker compose exec scanner alembic upgrade head`
  9. Cleanup: `@rm -rf /tmp/scanner-restore`
  10. `@echo "Restore complete. Run 'make migrate' if schema was updated."`
- `package: ## Create distributable archive` --
  ```
  tar czf $(PACKAGE_NAME).tar.gz \
      --transform 's,^,$(PACKAGE_NAME)/,' \
      Dockerfile docker-compose.yml config.yml.example .env.example \
      README.md LICENSE CONTRIBUTING.md \
      src/ pyproject.toml alembic/ alembic.ini docs/
  @echo "Package created: $(PACKAGE_NAME).tar.gz"
  ```
- `clean: ## Remove build artifacts and caches` --
  ```
  find . -type d -name __pycache__ -exec rm -rf {} + 2>/dev/null || true
  find . -type f -name '*.pyc' -delete 2>/dev/null || true
  rm -f aipix-security-scanner-*.tar.gz
  ```
- `docker-multiarch: ## Build multi-arch images (amd64 + arm64)` --
  ```
  docker buildx create --name multiarch --driver docker-container --use 2>/dev/null || docker buildx use multiarch
  docker buildx build --platform linux/amd64,linux/arm64 \
      -t $(IMAGE_NAME):$(VERSION) \
      --output type=oci,dest=$(PACKAGE_NAME)-multiarch.tar .
  @echo "Multi-arch image saved: $(PACKAGE_NAME)-multiarch.tar"
  ```
- `docker-push: ## Push multi-arch images to registry` --
  ```
  docker buildx build --platform linux/amd64,linux/arm64 \
      -t $(REGISTRY)/$(IMAGE_NAME):$(VERSION) --push .
  ```
