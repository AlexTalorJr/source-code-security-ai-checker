@asynccontextmanager
async def lifespan(app: FastAPI):
    settings = ScannerSettings()
    engine = create_engine(settings.db_path)
    async with engine.begin() as conn:
        await conn.run_sync(Base.metadata.create_all)
    app.state.engine = engine
    app.state.session_factory = create_session_factory(engine)
    app.state.settings = settings
    yield
    await engine.dispose()

def create_app() -> FastAPI:
    app = FastAPI(title="aipix-security-scanner", version="0.1.0", lifespan=lifespan)
    app.include_router(api_router, prefix="/api")

      - `__init__(self)`: creates `self._queue: asyncio.Queue[int] = asyncio.Queue()` and `self._current_scan_id: int | None = None`
      - `async def enqueue(self, scan_id: int)`: puts scan_id on queue
      - `@property def current_scan_id(self) -> int | None`: returns _current_scan_id
      - `async def worker(self, app: FastAPI)`: infinite loop that:
        1. `scan_id = await self._queue.get()`
        2. Sets `self._current_scan_id = scan_id`
        3. Gets session from `app.state.session_factory`
        4. Updates scan status to "running" in DB
        5. Reads scan record to get target_path/repo_url/branch
        6. Calls `run_scan(app.state.settings, target_path=..., repo_url=..., branch=...)`
        7. On success: updates DB record with results from ScanResultSchema (all counts, gate_passed, timestamps, etc.)
        8. On exception: updates status to "failed" and sets error_message
        9. Finally: `self._current_scan_id = None`, `self._queue.task_done()`
        10. After completion (success or fail): calls `await notify_scan_complete(...)` from `scanner.notifications.service` inside try/except (import may fail if notifications module doesn't exist yet -- catch ImportError gracefully)
      - `async def recover_stuck_scans(self, app: FastAPI)`: queries scans with status "running" or "queued", resets "running" to "queued", re-enqueues all
