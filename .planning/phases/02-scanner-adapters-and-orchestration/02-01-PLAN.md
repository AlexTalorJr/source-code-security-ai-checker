```python
class ScannerError(Exception):
    """Base exception for scanner errors."""

class ScannerTimeoutError(ScannerError):
    """Raised when a scanner tool exceeds its timeout."""
    def __init__(self, tool_name: str, timeout: int):
        self.tool_name = tool_name
        self.timeout = timeout
        super().__init__(f"{tool_name} timed out after {timeout}s")

class ScannerExecutionError(ScannerError):
    """Raised when a scanner tool fails to execute."""
    def __init__(self, tool_name: str, message: str, returncode: int | None = None):
        self.tool_name = tool_name
        self.returncode = returncode
        super().__init__(f"{tool_name} failed: {message}")

class GitCloneError(ScannerError):
    """Raised when git clone fails."""
```

1. Create `src/scanner/core/git.py` with two async functions:

`async def clone_repo(repo_url: str, branch: str, shallow: bool = True, git_token: str | None = None) -> str`:
- Creates temp directory with `tempfile.mkdtemp(prefix="scanner-clone-")`
- If `git_token` and `repo_url.startswith("https://")`: creates a `.git-askpass` script in tmpdir that echoes the token, sets `GIT_ASKPASS` env var, sets script executable with `os.chmod(path, 0o700)`
- Builds command: `["git", "clone", "--branch", branch, "--single-branch"]`
- If `shallow`: appends `["--depth", "1"]`
- Appends `[repo_url, tmpdir]`
- Runs via `asyncio.create_subprocess_exec` with custom env, PIPE for stdout/stderr
- On non-zero returncode: cleans up tmpdir with `shutil.rmtree(tmpdir, ignore_errors=True)`, raises `GitCloneError(f"git clone failed (exit {proc.returncode}): {stderr.decode()}")`
- Returns tmpdir path

`def cleanup_clone(clone_path: str) -> None`:
- Calls `shutil.rmtree(clone_path, ignore_errors=True)`
- Synchronous -- called in finally blocks
